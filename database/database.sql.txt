CREATE TABLE `hotel_management`.`account_data` ( `sr` INT NOT NULL AUTO_INCREMENT ,  `username` VARCHAR(15) NOT NULL ,  `password` VARCHAR(20) NOT NULL ,  `email` VARCHAR(50) NOT NULL ,  `mobile` BIGINT NOT NULL ,  `verified` TINYINT NOT NULL ,  `name` VARCHAR(30) NOT NULL ,    PRIMARY KEY  (`sr`)) ENGINE = InnoDB;
CREATE TABLE `hotel_management`.`booking` ( `sr` INT(11) NOT NULL AUTO_INCREMENT ,  `room_no` INT(11) NOT NULL ,  `check_in` DATE NOT NULL ,  `check_out` DATE NOT NULL ,  `room_type` INT(11) NOT NULL ,  `guest_no` INT(11) NOT NULL ,  `guest_id` INT(11) NOT NULL ,  `amount_paid` INT(11) NOT NULL ,    PRIMARY KEY  (`sr`)) ENGINE = InnoDB;
